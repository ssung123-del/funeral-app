    // --- 0. 초기화: 확실한 현지 시간 오늘 날짜 계산 ---

    window.onload = function() {

        const now = new Date();

        const year = now.getFullYear();

        const month = String(now.getMonth() + 1).padStart(2, '0');

        const day = String(now.getDate()).padStart(2, '0');

        const today = `${year}-${month}-${day}`;

        

        document.querySelectorAll('input[type="date"]').forEach(input => input.value = today);

    };


    // --- 1. 유틸리티 함수 ---

    function clean(text) { return text ? text.replace(/\s+/g, '') : ""; }

    

    function formatDate(dateStr) {

        if (!dateStr) return "";

        const [y, m, d] = dateStr.split('-').map(Number);

        const target = new Date(y, m - 1, d);

        const today = new Date();

        target.setHours(0,0,0,0); today.setHours(0,0,0,0);

        const weekDays = ['일','월','화','수','목','금','토'];

        return (target.getTime() === today.getTime()) ? `${d}일(오늘)` : `${d}일(${weekDays[target.getDay()]})`;

    }


    function combineDateTime(dateId, timeId) {

        const timeVal = clean(document.getElementById(timeId).value);

        if (!timeVal) return "";

        return `${formatDate(document.getElementById(dateId).value)}${timeVal}`;

    }


    function getByteLength(s) {

        let b = 0;

        for (let i = 0; i < s.length; i++) {

            const c = s.charCodeAt(i);

            b += (c >> 7) ? 2 : 1;

        }

        return b;

    }


    function buildPersonName(n, t, spouseN, spouseT) {

        if (!n) return "";

        if (!spouseN) return `${n}${t}`;

        if (t === spouseT) return `${n}(${spouseN})${t}`;

        return `${n}${t}(${spouseN}${spouseT})`;

    }

            let headerText = `교회장-${mainDistrict}`;


            // 고인 이름 (故 추가)

            let nameStr = buildPersonName("故"+mName, mTitle, mSpouse, mSpouseTitle);

            body = `${nameStr}소천`; 


            // 겹장 루프 시작 (항상 체크)

            const cards = document.querySelectorAll('.joint-card');

            

            cards.forEach(card => {

                let jType = card.querySelector('.j-type').value;

                let jDist = clean(card.querySelector('.j-district').value).replace(/교구/g, '');

                let jName = clean(card.querySelector('.j-name').value);

                let jTitle = card.querySelector('.j-title').value;

                let jSpouse = clean(card.querySelector('.j-spouse').value);

                let jSpouseTitle = card.querySelector('.j-spouse-title').value;

                let jRel = clean(card.querySelector('.j-rel').value);


                // 필수값(교구명, 이름)이 있을 때만 추가

                if (jDist && jName) {

                    headerText += `/${jType}-${jDist}`;

                    let jNameStr = buildPersonName(jName, jTitle, jSpouse, jSpouseTitle);

                    body += `/${jNameStr}${jRel}`;

                }

            });

            

            header = `[${headerText}]`;


            // 교회장 일정

            let wiro = combineDateTime('date_wiro_church', 'time_wiro_church');

            let ipgwan = combineDateTime('date_ipgwan', 'time_ipgwan');

            let sendoff = combineDateTime('date_sendoff', 'time_sendoff');


            if (wiro) body += `/위로:${wiro}교회출`;

            if (ipgwan) body += `/입관:${ipgwan}교회출`;

            if (sendoff) body += `/천국환송:${sendoff}교회출`;

        }


        // 최종 조립

        let result = `${header}${body}/${location}`;


        const resultArea = document.getElementById('resultArea');

        const output = document.getElementById('output');

        const byteVal = document.getElementById('byteVal');

        const smsType = document.getElementById('smsType');


        output.innerText = result;

        const bytes = getByteLength(result);

        byteVal.innerText = bytes;


        if (bytes <= 90) {

            smsType.innerText = "SMS";

            smsType.className = "byte-badge sms";

        } else {

            smsType.innerText = "LMS";

            smsType.className = "byte-badge lms";

        }


        resultArea.style.display = 'block';

        resultArea.scrollIntoView({ behavior: 'smooth', block: 'end' });

    }
